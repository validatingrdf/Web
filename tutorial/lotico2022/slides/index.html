RDF

prefix :      <http://example.org/>
prefix rdfs:  <http://www.w3.org/2000/01/rdf-schema#>
prefix xsd:   <http://www.w3.org/2001/XMLSchema#>
prefix rdf:   <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

:weavingWeb rdf:type :Book ;
            :title  "Weaving the Web" ;
            :code   "isbn10:0-06-251587-X" ;
            :code   "isbn13:978-0-06-251587-2" ;
            :author :timbl .
:timbl  rdf:type    :Human ;
        :birthPlace :london ;
        rdfs:label  "Tim Berners-Lee" ;
        :birthDate  "1955-06-08"^^xsd:date ;
        :knows      _:1   .
:london rdf:type :City, :Metropolis ;
        :country :UK .
_:1     :birthPlace :Spain .
